\question[10] Let $p$ and $q$ be the propositions
 \begin{itemize}
    \item[$p$:] I went hiking this week.
    \item[$q$:] I have a sunburn.
 \end{itemize}
 Express each of these propositions as an English sentence.
    \begin{parts}
    \part $\neg q$
    \part $p \vee q$
    \part $q \rightarrow p$
    \part $\neg p \wedge \neg q$
    \part $p \leftrightarrow q$
    \end{parts}
    \begin{solution}
        \begin{parts}
        \part It is not the case that I have a sunburn.
        \part I went hiking this week or I have a sunburn.
        \part If I have a sunburn then I went hiking this week.
        \part I did not go hiking this week and I do not have a sunburn.
        \part I went hiking this week if and only if I have a sunburn.
        \end{parts}
    \end{solution}
